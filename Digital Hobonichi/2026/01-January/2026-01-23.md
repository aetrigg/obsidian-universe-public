let file = app.workspace.getActiveFile();

  if (line.trim().startsWith("| Cost")) {

  }

  if (inTable) {
    if (!line.trim().startsWith("|") || line.trim().startsWith("| ----")) continue;

    let cells = line.split("|").map(cell => cell.trim());

    if (!isNaN(cost)) {
